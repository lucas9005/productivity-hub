import { Component, signal, computed } from '@angular/core';
import { CommonModule } from '@angular/common';

/**
 * @description Example feature component demonstrating the use of Angular Signals
 * for feature-specific state management.
 *
 * This component implements a simple counter with increment and decrement functionality
 * using Angular's Signal API for reactive state management. It also demonstrates
 * computed values with a doubled count value.
 *
 * @usageNotes
 * The component can be used in templates as follows:
 * ```html
 * <app-example-feature></app-example-feature>
 * ```
 *
 * Example of integration with parent component:
 * ```typescript
 * @Component({
 *   selector: 'app-parent',
 *   template: `
 *     <div>
 *       <h1>Parent Component</h1>
 *       <app-example-feature></app-example-feature>
 *     </div>
 *   `
 * })
 * export class ParentComponent {}
 * ```
 *
 * The component provides:
 * - A counter display
 * - Increment/decrement buttons
 * - A computed doubled value
 *
 * @example
 * ```typescript
 * // Access the counter value in the template
 * {{ count() }}
 *
 * // Access the doubled value
 * {{ doubledCount() }}
 * ```
 */
@Component({
  selector: 'app-example-feature',
  standalone: true,
  imports: [CommonModule],
  template: `
    <div class="p-4">
      <h2 class="text-2xl font-bold mb-4">Example Feature</h2>
      <div class="flex items-center gap-4">
        <button class="px-4 py-2 bg-blue-500 text-white rounded" (click)="decrement()">-</button>
        <span class="text-xl">{{ count() }}</span>
        <button class="px-4 py-2 bg-blue-500 text-white rounded" (click)="increment()">+</button>
      </div>
      <div class="mt-4">
        <p>Doubled value: {{ doubledCount() }}</p>
      </div>
    </div>
  `,
})
export class ExampleFeatureComponent {
  /**
   * @description Signal representing the current count value.
   * Protected to allow template access while maintaining encapsulation.
   *
   * @default 0
   */
  protected readonly count = signal(0);

  /**
   * @description Computed signal that returns double the current count value.
   * Updates automatically when count changes.
   *
   * @returns The current count multiplied by 2
   */
  readonly doubledCount = computed(() => this.count() * 2);

  /**
   * @description Increments the count value by 1 using immutable update.
   *
   * @example
   * ```typescript
   * <button (click)="increment()">+</button>
   * ```
   */
  increment(): void {
    this.count.update((c) => c + 1);
  }

  /**
   * @description Decrements the count value by 1, but never below 0
   * Uses Math.max to ensure the count never goes below 0
   *
   * @remarks
   * This method demonstrates input validation and constraints on signal updates.
   * The count will never go below 0, showing how to implement business rules
   * in signal updates.
   *
   * @example
   * ```typescript
   * // In template
   * <button (click)="decrement()">-</button>
   *
   * // In component
   * this.decrement();
   * ```
   */
  decrement(): void {
    this.count.update((c) => Math.max(0, c - 1));
  }
}
