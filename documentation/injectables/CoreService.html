import { Injectable } from '@angular/core';

/**
 * @description Core service providing application-wide functionality and initialization management.
 *
 * @usageNotes
 * Inject into components/services to access core functionality:
 * ```typescript
 * constructor(private coreService: CoreService) {
 *   if (!this.coreService.isInitialized) {
 *     this.coreService.initialize();
 *   }
 * }
 * ```
 */
@Injectable({
  providedIn: 'root',
})
export class CoreService {
  /**
   * @description Flag indicating whether the application has been initialized.
   * Set to true after successful initialization to prevent duplicate initialization.
   *
   * @default false
   */
  isInitialized = false;

  /**
   * @description Creates an instance of CoreService.
   * This service follows the singleton pattern and is provided at the root level.
   *
   * @remarks
   * - This service should be instantiated only once during the application lifecycle
   * - The constructor is intentionally empty as initialization logic is handled by the initialize() method
   * - Dependencies should be injected here if needed for core functionality
   *
   * @example
   * ```typescript
   * // Inject dependencies if needed
   * constructor(
   *   private store: Store,
   *   private router: Router
   * ) {}
   * ```
   */
  constructor() {}

  /**
   * @description Initializes the application by setting up core functionality.
   * This method should be called only once during the application bootstrap process.
   *
   * The method:
   * 1. Checks if the service is already initialized
   * 2. Sets up core functionality if needed
   * 3. Marks the service as initialized
   *
   * @throws {Error} If called when the application is already initialized
   * @returns void
   *
   * @example
   * Basic usage:
   * ```typescript
   * // In your root component
   * ngOnInit() {
   *   if (!this.coreService.isInitialized) {
   *     this.coreService.initialize();
   *   }
   * }
   * ```
   *
   * Error handling:
   * ```typescript
   * try {
   *   this.coreService.initialize();
   * } catch (error) {
   *   console.error('Failed to initialize core service:', error);
   *   // Handle initialization failure
   * }
   * ```
   */
  initialize(): void {
    if (this.isInitialized) {
      throw new Error('CoreService has already been initialized');
    }
    this.isInitialized = true;
  }
}
